rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /replays/{replayId} {
      allow read: if true;
      allow create: if request.resource.data.keys().hasAll(['initialState', 'moves', 'finalScore', 'createdAt'])
                    && request.resource.data.initialState.snake is list
                    && request.resource.data.initialState.apple is map
                    && request.resource.data.moves is list
                    && request.resource.data.finalScore is number
                    && request.resource.data.createdAt == request.time;
      allow update, delete: if false;
    }
  }
}
